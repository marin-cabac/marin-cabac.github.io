// Function to generate unique numbers across multiple lists function generateUniqueNumbersAcrossLists(totalLists, count, min, max) { const rangeSize = max - min + 1; if (totalLists * count > rangeSize) { // Throw an error if the range is too small for the request console.error( `Cannot generate ${totalLists * count} unique numbers in range ${min}-${max}.` ); return []; // Return an empty array to prevent issues } const usedNumbers = new Set(); const allLists = []; for (let i = 0; i < totalLists; i++) { const currentList = []; while (currentList.length < count) { const num = Math.floor(Math.random() * (max - min + 1)) + min; if (!usedNumbers.has(num)) { usedNumbers.add(num); currentList.push(num); } } allLists.push(currentList); } return allLists; } // Generate EUROMILLIONS numbers function generateEuroMillions() { const mainContainer = document.getElementById('euro-main'); const luckyContainer = document.getElementById('euro-lucky'); // Main Numbers: Unique across all lists const mainNumbersLists = generateUniqueNumbersAcrossLists(5, 5, 1, 50); if (!mainNumbersLists.length) return; // Stop if generation failed // Lucky Numbers: Unique across all lists const luckyNumbersLists = generateUniqueNumbersAcrossLists(5, 5, 1, 12); if (!luckyNumbersLists.length) return; // Stop if generation failed for (let i = 0; i < 5; i++) { const mainList = document.createElement('div'); mainList.className = 'list'; mainList.textContent = `Main Numbers: ${mainNumbersLists[i].join(', ')}`; const luckyList = document.createElement('div'); luckyList.className = 'list'; luckyList.textContent = `Lucky Numbers: ${luckyNumbersLists[i].join(', ')}`; mainContainer.appendChild(mainList); luckyContainer.appendChild(luckyList); } } // Generate LOTO numbers function generateLoto() { const mainContainer = document.getElementById('loto-main'); const specialContainer = document.getElementById('loto-special'); // Main Numbers: Unique across all lists const mainNumbersLists = generateUniqueNumbersAcrossLists(5, 5, 1, 49); if (!mainNumbersLists.length) return; // Stop if generation failed // Special Numbers: Unique across all lists const specialNumbersLists = generateUniqueNumbersAcrossLists(5, 5, 1, 10); if (!specialNumbersLists.length) return; // Stop if generation failed for (let i = 0; i < 5; i++) { const mainList = document.createElement('div'); mainList.className = 'list'; mainList.textContent = `Main Numbers: ${mainNumbersLists[i].join(', ')}`; const specialList = document.createElement('div'); specialList.className = 'list'; specialList.textContent = `Special Numbers: ${specialNumbersLists[i].join(', ')}`; mainContainer.appendChild(mainList); specialContainer.appendChild(specialList); } } // Generate all numbers on page load generateEuroMillions(); generateLoto();
